// Reference solution for Chapter 8: Token Bucket.
package main

import (
	"time"
)

type TokenBucket struct {
	rate     float64
	capacity float64
	tokens   float64
	lastTime time.Time
}

func NewTokenBucket(rate float64, capacity int) *TokenBucket {
	return &TokenBucket{
		rate:     rate,
		capacity: float64(capacity),
		tokens:   float64(capacity),
		lastTime: time.Time{},
	}
}

func (b *TokenBucket) Allow(now time.Time) bool {
	if b.lastTime.IsZero() {
		b.lastTime = now
	} else {
		elapsed := now.Sub(b.lastTime).Seconds()
		if elapsed > 0 {
			b.tokens += b.rate * elapsed
			if b.tokens > b.capacity {
				b.tokens = b.capacity
			}
			b.lastTime = now
		}
	}

	if b.tokens >= 1 {
		b.tokens -= 1
		return true
	}
	return false
}

func main() {}
