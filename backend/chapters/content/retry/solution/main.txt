// Reference solution for Chapter 10: HTTP Retry & Backoff.
package main

import (
	"context"
	"net/http"
	"time"
)

func DoWithRetry(ctx context.Context, url string, maxRetries int, baseDelay time.Duration) (*http.Response, error) {
	client := &http.Client{Timeout: 5 * time.Second}

	for attempt := 0; attempt <= maxRetries; attempt++ {
		if err := ctx.Err(); err != nil {
			return nil, err
		}

		req, err := http.NewRequestWithContext(ctx, http.MethodGet, url, nil)
		if err != nil {
			return nil, err
		}
		resp, err := client.Do(req)
		if err != nil {
			// transport error — retry unless last attempt
			if attempt == maxRetries {
				return nil, err
			}
		} else if resp.StatusCode >= 200 && resp.StatusCode < 300 {
			return resp, nil
		} else {
			_ = resp.Body.Close()
		}

		if attempt == maxRetries {
			// exhausted; return the last response so the caller sees the status
			if resp != nil {
				return resp, nil
			}
			return nil, ctx.Err()
		}

		delay := baseDelay * (1 << attempt) // baseDelay * 2^attempt
		select {
		case <-ctx.Done():
			return nil, ctx.Err()
		case <-time.After(delay):
		}
	}
	return nil, nil
}

func main() {}
