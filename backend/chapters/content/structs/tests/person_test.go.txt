package main

import (
	"testing"
)

func TestNewPersonSetsFields(t *testing.T) {
	p := NewPerson("Alice", 30)
	if p == nil {
		t.Fatal("NewPerson returned nil")
	}
	if p.Name != "Alice" {
		t.Errorf("Name = %q, want %q", p.Name, "Alice")
	}
	if p.Age != 30 {
		t.Errorf("Age = %d, want 30", p.Age)
	}
	if p.Emails != nil {
		t.Errorf("Emails should be nil by default, got %v", p.Emails)
	}
}

func TestPersonString(t *testing.T) {
	p := NewPerson("Bob", 25)
	if got, want := p.String(), "Bob (25)"; got != want {
		t.Errorf("String() = %q, want %q", got, want)
	}
}

func TestPersonHasEmail(t *testing.T) {
	p := NewPerson("Carol", 40)
	p.Emails = []string{"carol@example.com", "carol@other.com"}

	cases := []struct {
		email string
		want  bool
	}{
		{"carol@example.com", true},
		{"CAROL@EXAMPLE.COM", true},
		{"carol@other.com", true},
		{"missing@example.com", false},
		{"", false},
	}
	for _, tc := range cases {
		if got := p.HasEmail(tc.email); got != tc.want {
			t.Errorf("HasEmail(%q) = %v, want %v", tc.email, got, tc.want)
		}
	}
}

func TestPersonHasEmailNil(t *testing.T) {
	p := NewPerson("Dave", 50)
	if p.HasEmail("anything@example.com") {
		t.Error("HasEmail should return false when Emails is nil")
	}
}
